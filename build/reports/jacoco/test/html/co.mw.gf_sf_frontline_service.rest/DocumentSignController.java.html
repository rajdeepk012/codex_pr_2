package co.mw.gf_sf_frontline_service.rest;

import co.mw.gf_sf_frontline_service.service.DocumentSignService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.*;
import org.springframework.web.bind.annotation.*;
import java.io.IOException;

import static co.mw.gf_sf_frontline_service.common.Constants.API_VERSION;

@RestController
@RequestMapping(API_VERSION+"/eSign")
public class DocumentSignController {

    private DocumentSignService documentSignService;
    @Autowired
    public DocumentSignController(DocumentSignService documentSignService){
        this.documentSignService=documentSignService;
    }

    @PostMapping(value="/contract", consumes="application/json")
    public ResponseEntity<Object> sendContractSignRequest(@RequestBody Object request){
        return documentSignService.sendSignRequest(request);
    }

    @PostMapping(value = "/remoteESigning")
    public ResponseEntity<Object> signAndRetrieveSignedDocument(
            @RequestBody Object request) throws IOException {
        return documentSignService.sendRemoteESignRequest(request);
    }
}
