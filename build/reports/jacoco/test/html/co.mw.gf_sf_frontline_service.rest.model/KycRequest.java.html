package co.mw.gf_sf_frontline_service.rest.model;

import lombok.*;

@AllArgsConstructor
@NoArgsConstructor
@Builder
@Getter
@Setter
@ToString
public class KycRequest {
    private PanKycRequest pan;

    private AadharKycRequest aadhaar;
    private UtilityBillKycRequest utilityBill;

    private MobileKycRequest mobile;
    public PanKycRequest getPan() {
        return pan;
    }

    public void setPan(PanKycRequest pan) {
        this.pan = pan;
    }

    public AadharKycRequest getAadhaar() {
        return aadhaar;
    }

    public void setAadhaar(AadharKycRequest aadhaar) {
        this.aadhaar = aadhaar;
    }

    public UtilityBillKycRequest getUtilityBill() {
        return utilityBill;
    }

    public void setUtilityBill(UtilityBillKycRequest utilityBill) {
        this.utilityBill = utilityBill;
    }

    public MobileKycRequest getMobile() {
        return mobile;
    }

    public void setMobile(MobileKycRequest mobile) {
        this.mobile = mobile;
    }
}
